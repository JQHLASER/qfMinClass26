
1> ManualResetEvent方法	  (后台线程)
		ManualResetEvent evt = new ManualResetEvent(false);	
		// 等待线程
		evt.WaitOne();   // 阻塞等待，不耗 CPU

		// 设置线程
		evt.Set();       // 通知 ,等待结束
