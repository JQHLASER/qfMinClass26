代码设置列标题的显示/隐藏
   1> 通过设置列标题的高度为 0 来隐藏：
         if (listView1.View is GridView gridView)
          {
              // 设置列标题高度为0
              gridView.ColumnHeaderHeight = 0;
          }
   2> 通过创建空样式来隐藏
      // 假设你的ListView名为listView1
      if (listView1.View is GridView gridView)
      {
          // 创建一个空样式，用于隐藏列标题
          Style emptyStyle = new Style();
          emptyStyle.TargetType = typeof(GridViewColumnHeader);
          
          // 应用样式到GridView
          gridView.ColumnHeaderContainerStyle = emptyStyle;
      }
