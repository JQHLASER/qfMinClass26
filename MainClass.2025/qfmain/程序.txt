自定义依赖属性

 // 依赖属性的标识符....必须要申明后才能被Binding绑定
 public static readonly DependencyProperty ui_TextIntProperty =
     DependencyProperty.Register(
         "ui_TextInt",  // 属性名称
         typeof(int),   // 属性类型
         typeof(ui_TextBox),  // 拥有者类型
         new PropertyMetadata(0)// 默认值
         );

   private string TextUi_ = "TextBox";
        /// <summary>
        /// 显示文本
        /// </summary>
        [Description("显示文本")]
        public string ui_Text
        {
            get
            {
                return (string)GetValue (ui_TextProperty);
            }
            set
            {
                TextUi_ = value;
                SetValue(ui_TextProperty, value);
            }
        }





若自定义控件的构造函数或 XAML 中设置了DataContext = this，会导致外部的DataContext无法继承，必须显式指定Source：
例  ui_Text ="{Binding DataContext.text, RelativeSource={RelativeSource AncestorType=Window}  }"  //绑定自身

<!-- 场景1：源是某个命名控件 -->
<local:MyCustomControl MyValue="{Binding Text, Source={x:Reference Name=MyTextBox}}" />
<!-- 场景2：源是静态资源 -->
<local:MyCustomControl MyValue="{Binding Age, Source={StaticResource UserData}}" />
<!-- 场景3：源是父元素的DataContext -->
<local:MyCustomControl MyValue="{Binding DataContext.ViewModelProperty, RelativeSource={RelativeSource AncestorType=Window}}" />
