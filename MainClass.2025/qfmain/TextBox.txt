 // 自定义方法：限制输入为数字
        public void RestrictToNumbers()
        {
            this.PreviewTextInput += (sender, e) =>
            {
                e.Handled = !IsNumber(e.Text);
            };
            
            DataObject.AddPastingHandler(this, (sender, e) =>
            {
                if (!IsNumber(e.DataObject.GetData(DataFormats.Text) as string))
                {
                    e.CancelCommand();
                }
            });
        }
