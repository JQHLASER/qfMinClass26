
一般控制台会在运行后立马关闭,这种方法可以一直运行,


 static ManualResetEvent _quitEvent = new ManualResetEvent(false);
 static void Main(string[] args)
 {
    
     // 注册Ctrl+C事件处理
     Console.CancelKeyPress += (sender, e) =>
     {
         e.Cancel = true; // 取消默认处理（默认会直接终止程序）
         _quitEvent.Set(); // 通知程序退出
     };
     // 等待退出信号
     _quitEvent.WaitOne();
 }
    


控制台运行后不关闭的方法
        Console.ReadLine(); // 这样程序会在输出完毕后等待用户输入，直到按下 Enter 键后才会退出。
        Console.ReadLine(); //但是 Console.ReadKey() 会等待用户按下任意键才会退出。
       
       // 无限循环，程序将一直保持运行状态
        while (true)
        {
            // 可以在这里做一些处理，或者让程序做一些等待等操作
            System.Threading.Thread.Sleep(1000);  // 每秒钟循环一次，避免占用过多CPU资源
        }


让窗体不显示
        // 导入 Windows API 函数
        [DllImport("kernel32.dll")]
        private static extern IntPtr GetConsoleWindow();

        [DllImport("user32.dll")]
        private static extern bool ShowWindow(IntPtr hWnd, int nCmdShow);

        private const int SW_HIDE = 0;     // 隐藏窗口
        private const int SW_SHOW = 5;     // 显示窗口（用于调试）

       
       //设置显示或不显示窗口
        IntPtr consoleHandle = GetConsoleWindow();
        ShowWindow(consoleHandle, SW_HIDE);
