        {

        };

        //事件触发时处理
        void On_Clear()
        {
            this.Item = new string[0];

            this.uiComboBox1.Invalidate();
        }

txt保存,生产者线程写入,消费者线程保存
    using (var writer = new StreamWriter("largefile.txt", false, Encoding.UTF8, 65536))
    {
        var queue = new BlockingCollection<string>();
        // 生产者线程生成数据
        Task.Run(() =>
        {
            foreach (var item in data)
                queue.Add(item);
            queue.CompleteAdding();
        });
        // 消费者线程写入文件
        foreach (var line in queue.GetConsumingEnumerable())
            writer.WriteLine(line);
    }


txt保存,二进制写入
    using (var stream = new FileStream("largefile.txt", FileMode.Create, FileAccess.Write, FileShare.None, 65536))
    {
        foreach (var line in data)
        {
            var bytes = Encoding.UTF8.GetBytes(line + Environment.NewLine);
            stream.Write(bytes, 0, bytes.Length);
        }
    }
